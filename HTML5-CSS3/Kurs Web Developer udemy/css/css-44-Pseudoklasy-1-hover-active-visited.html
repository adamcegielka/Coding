<style>
    a {
        font-weight: bold;
    }

/* gdy kursor jest nad elementem zostaje czciona czerwona */
    a:hover {
        color: red;
    }

/* gdy link jest kliknięty zostaje czciona zielina */
    a:active {
        color: green;
    }

/* gdy link był już odwiedzony */
    a:visited {
        color: yellowgreen;
    }
</style>
